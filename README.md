# ========== 账号管理 ==========
shorebird login                    # 登录
shorebird logout                   # 登出
shorebird account                  # 查看账号信息

# ========== 项目管理 ==========
shorebird init                     # 初始化项目
shorebird doctor                   # 检查环境
shorebird apps list                # 列出所有应用

# ========== 发布管理 ==========
shorebird release android          # 发布 Android
shorebird release ios              # 发布 iOS
shorebird releases list            # 列出所有发布

# ========== 补丁管理 ==========
shorebird patch android            # 创建 Android 补丁
shorebird patch ios                # 创建 iOS 补丁
shorebird patches list             # 列出所有补丁

# ========== 预览测试 ==========
shorebird preview                  # 本地预览补丁效果

# ========== 调试 ==========
shorebird build apk --verbose      # 详细日志构建
